public class Text1018 {
	public static void main(String[] args) {
		System.out.println("\t將3的倍數蓋掉的99乘法表");
		for(int i=1;i<10;i+=3){
			System.out.println("\t"+i+"\t"+(i+1)+"\t"+(i+2));
			System.out.println("\t======\t======\t======");
			for(int j=1;j<10;j++){
				if(j%3==0){
					System.out.println("\t●●●\t●●●\t●●●");
				}else{
					System.out.println("\t"+i+"*"+j+"="+i*j + "\t"+(i+1)+"*"+j+"="+((i+1)*j) + "\t●●●");
				}
				
			}
			System.out.println();
		}
	}
}
