import java.io.*;		//匯入IO函式庫

public class Text1004{
	public static void main(String args[])throws IOException{
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		//宣告BufferedReader為br,並且br的值為鍵盤輸入的值

		System.out.println("請輸入兩數:");
		
		String str1 = br.readLine();		//宣告新的字串,值從鍵盤輸入的一行指定
		String str2 = br.readLine();		//宣告新的字串,值從鍵盤輸入的一行指定
		System.out.print("未經轉換str1+str2的結果:");
		System.out.println(str1+str2);		//顯示未轉換前的字串相加結果
		
		int num1 = Integer.parseInt(str1);	//將str1字串轉換為INT的值給新變數num1
		int num2 = Integer.parseInt(str2);	//將str2字串轉換為INT的值給新變數num2
		System.out.print("經Integer轉換str1+str2的結果:");
		System.out.println(num1+num2);		//顯示轉換後整數相加的結果
		
		float f1 = Float.parseFloat(str1);	//將str1字串轉換為Float的值給新變數f1
		float f2 = Float.parseFloat(str2);	//將str1字串轉換為Float的值給新變數f2
		System.out.print("經Float轉換str1+str2的結果:");
		System.out.println(f1+f2);		//顯示轉換後浮點數相加的結果
		
		str1 = String.valueOf(f1);		//將str1的值更改為f1的值
		str2 = String.valueOf(f2);		//將str2的值更改為f2的值
		System.out.print("f1,f2經String.valueOf轉換str1+str2的結果:");
		System.out.println(str1+str2);		//顯示str1,str2的值
	}
}
