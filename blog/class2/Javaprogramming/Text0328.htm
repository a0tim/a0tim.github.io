import java.awt.Cursor;
import java.awt.event.KeyAdapter;
import java.awt.event.KeyEvent;
import java.awt.event.MouseAdapter;
import java.awt.event.MouseEvent;

import javax.swing.ImageIcon;
import javax.swing.JApplet;
import javax.swing.JLabel;
import javax.swing.plaf.basic.BasicInternalFrameTitlePane.IconifyAction;


public class Text0328 extends JApplet{
	JLabel JL1;
	JLabel JL2 = new JLabel();
	JLabel JL3;
	ImageIcon img1 = new ImageIcon("C:\\01.jpg");
	ImageIcon img2 = new ImageIcon("C:\\02.jpg");
	ImageIcon img3 = new ImageIcon("C:\\03.jpg");
	ImageIcon img4 = new ImageIcon("C:\\04.jpg");
	public void init(){
		setLayout(null);
		setSize(800,600);
		JL1 = new JLabel(img1);
		JL1.setBounds(50,10,90,125);
		JL1.addMouseListener(new MAction());
		JL2.setBounds(60,150,90,20);
		JL2.setText("王慶生  主任");
		JL3 = new JLabel(img4);
		JL3.setBounds(-100,-100,50,50);
		add(JL1);
		add(JL2);
		add(JL3);
		addKeyListener(new KAction());
		addMouseMotionListener(new MAction2());
		setCursor(Cursor.getPredefinedCursor(Cursor.HAND_CURSOR));
	}
	public class MAction extends MouseAdapter{
		public void mouseEntered(MouseEvent e){
			JL1.setIcon(img2);
			JL2.setText("黃信貿  老師");
		}
		public void mouseExited(MouseEvent e){
			JL1.setIcon(img1);
			JL2.setText("王慶生  主任");
		}
		public void mouseClicked(MouseEvent e){
			JL1.setIcon(img3);
			JL2.setText("陳炯良  老師");
		}
	}
	public class MAction2 extends MouseAdapter{
		public void mouseMoved(MouseEvent e){
			JL3.setLocation(JL3.getLocation().x + ((e.getPoint().x-JL3.getLocation().x)/20),JL3.getLocation().y + ((e.getPoint().y-JL3.getLocation().y)/20));
		}
	}
	public class KAction extends KeyAdapter{
		public void keyPressed(KeyEvent e){
			int k = e.getKeyCode();
			switch(k){
				case KeyEvent.VK_UP:
					JL1.setLocation(JL1.getLocation().x,JL1.getLocation().y -= 3);
					JL2.setLocation(JL2.getLocation().x,JL2.getLocation().y -= 3);
					break;
				case KeyEvent.VK_DOWN:
					JL1.setLocation(JL1.getLocation().x,JL1.getLocation().y += 3);
					JL2.setLocation(JL2.getLocation().x,JL2.getLocation().y += 3);
					break;
				case KeyEvent.VK_LEFT:
					JL1.setLocation(JL1.getLocation().x -= 3,JL1.getLocation().y);
					JL2.setLocation(JL2.getLocation().x -= 3,JL2.getLocation().y);
					break;
				case KeyEvent.VK_RIGHT:
					JL1.setLocation(JL1.getLocation().x += 3,JL1.getLocation().y);
					JL2.setLocation(JL2.getLocation().x += 3,JL2.getLocation().y);
					break;
				default:
			}
		}
	}
}
