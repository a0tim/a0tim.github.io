import java.io.*;		//匯入IO函式庫

public class Text1011{
	public static void main(String args[])throws IOException{
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		//宣告BufferedReader為br,並且br的值為鍵盤輸入的值

		int end = 0;		//宣告變數end用來當作do-while迴圈終止條件
		String a1="",b1="";	//迴圈外先宣告用來當驗證的帳號密碼字串
					//目的在於避免迴圈內無限的重複宣告
		
		System.out.println("\t====註冊====");
		System.out.print("\t請輸入帳號：");
		String a = br.readLine();		//預設帳號的值
		System.out.print("\t請輸入密碼：");
		String b = br.readLine();		//預設密碼的值
			
		do{
			System.out.println("\n\t====驗證====");
			System.out.print("\t請輸入帳號：");
			a1 = br.readLine();		//輸入帳號的值
			System.out.print("\t請輸入密碼：");
			b1 = br.readLine();		//輸入密碼的值
			if(a.equals(a1)){		//判斷輸入帳號是否符合註冊時輸入的值
				if(b.equals(b1)){	//判斷輸入密碼是否符合註冊時輸入的值
					end = 1;	//帳號密碼皆正確,將end的值設為1,終止迴圈
					System.out.println("\n\t!!驗證成功!!");
				}else{			//帳號正確,密碼輸入錯誤
					System.out.println("\t帳號正確!密碼錯誤!");
				}
			}else{				//帳號錯誤(並無判斷密碼)
				System.out.println("\t帳號錯誤!");
			}
		}while(end!=1);				//迴圈終止條件
	}
}
