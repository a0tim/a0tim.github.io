import java.io.*;

class Text1119_2{
	public static void main(String args[])throws IOException{
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		System.out.println("請輸入查詢年分：");
		int Year = Integer.parseInt(br.readLine());
		int Count = Year/4 - Year/100 + Year/400;
		int Date_start = ((Year-1)*365 + Count)%7;
		int Date[] = new int[42];
		if((Year%4) == 0 || (Year%400) == 0 && (Year%100) != 0){
			System.out.println("年分："+Year);
			for(int i=1;i<13;i++){
				int c = 1;
				if(i==1 || i==3 || i==5 || i==7 || i==8 || i==10 || i==12){
					for(int x=Date_start;x<42;x++,c++){
						Date[x] = c;
						if(c == 31){
							Date_start = (x+1)%7;
							break;
						}
					}
				}else if(i==4 || i==6 || i==9 || i==11){
					for(int x=Date_start;x<42;x++,c++){
						Date[x] = c;
						if(c == 30){
							Date_start = (x+1)%7;
							break;
						}
					}
				}else{
					for(int x=Date_start;x<42;x++,c++){
						Date[x] = c;
						if(c == 29){
							Date_start = (x+1)%7;
							break;
						}
					}
				}
				System.out.println(i+"月  "+Year);
				System.out.println("日\t一\t二\t三\t四\t五\t六");
				for(int j=0;j<6;j++){
					for(int k=j*7;k<((j*7)+7);k++){
						if(Date[k] == 0){
							System.out.print("\t");
							continue;
						}
						System.out.print(Date[k]+"\t");
					}
					System.out.println();
				}
				System.out.println();
				for(int x=0;x<42;x++){
					Date[x] = 0;
				}
			}
		}else{
			Date_start++;
			System.out.println("年分："+Year);
			for(int i=1;i<13;i++){
				int c = 1;
				if(i==1 || i==3 || i==5 || i==7 || i==8 || i==10 || i==12){
					for(int x=Date_start;x<42;x++,c++){
						Date[x] = c;
						if(c == 31){
							Date_start = (x+1)%7;
							break;
						}
					}
				}else if(i==4 || i==6 || i==9 || i==11){
					for(int x=Date_start;x<42;x++,c++){
						Date[x] = c;
						if(c == 30){
							Date_start = (x+1)%7;
							break;
						}
					}
				}else{
					for(int x=Date_start;x<42;x++,c++){
						Date[x] = c;
						if(c == 28){
							Date_start = (x+1)%7;
							break;
						}
					}
				}
				System.out.println(i+"月  "+Year);
				System.out.println("日\t一\t二\t三\t四\t五\t六");
				for(int j=0;j<6;j++){
					for(int k=j*7;k<((j*7)+7);k++){
						if(Date[k] == 0){
							System.out.print("\t");
							continue;
						}
						System.out.print(Date[k]+"\t");
					}
					System.out.println();
				}
				System.out.println();
				for(int x=0;x<42;x++){
					Date[x] = 0;
				}
			}
		}
	}
}
