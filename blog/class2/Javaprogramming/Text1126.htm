import java.util.*;

public class Text1126{
	public static void main(String args[]){
		Scanner sc = new Scanner(System.in);
		
		student s1 = new student(1871,"黃騰嶢","資工一A",100,100,100);
		teacher t1 = new teacher(1234,"黃信貿","資工一A");
		
		s1.printinfo();
		t1.printinfo();
	}
}

class teacher{
	int id;
	String name;
	String classname;
	
	public teacher(int a,String b,String c){
		this.id = a;
		this.name = b;
		this.classname = c;
	}
	
	public void printinfo(){
		System.out.println("====老師====");
		System.out.println("ID："+this.id);
		System.out.println("姓名："+this.name);
		System.out.println("班級："+this.classname);
	}
}

class student{
	int id;
	String name;
	String classname;
	int chi;
	int eng;
	int mat;
	
	public student(int a,String b,String c,int d,int e,int f){
		this.id = a;
		this.name = b;
		this.classname = c;
		this.chi = d;
		this.eng = e;
		this.mat = f;
	}
	
	public void printinfo(){
		System.out.println("====學生====");
		System.out.println("ID："+this.id);
		System.out.println("姓名："+this.name);
		System.out.println("班級："+this.classname);
		System.out.println("國文："+this.chi);
		System.out.println("英文："+this.eng);
		System.out.println("數學："+this.mat);
	}
}
