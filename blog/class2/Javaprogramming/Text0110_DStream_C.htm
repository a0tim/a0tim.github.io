import java.io.BufferedReader;
import java.io.DataInputStream;
import java.io.DataOutputStream;
import java.io.IOException;
import java.io.InputStreamReader;
import java.net.Socket;
import java.net.UnknownHostException;

public class Text0110_DStream_C {
	public static void main(String[] args) {
		try {
			Socket SC = new Socket("127.0.0.1",12345);
			System.out.println("Connect success!! Server:"+SC.getInetAddress());
			DataInputStream DI = new DataInputStream(SC.getInputStream());
			DataOutputStream DO = new DataOutputStream(SC.getOutputStream());
			BufferedReader BR = new BufferedReader(new InputStreamReader(System.in));
			System.out.println(DI.readUTF());
			System.out.print(DI.readUTF());
			String ClientName = BR.readLine();
			DO.writeUTF(ClientName);
			System.out.println(DI.readUTF());
			while(true){
				System.out.println("Server:"+DI.readUTF());
				System.out.print(ClientName+":");
				DO.writeUTF(BR.readLine());
			}
		} catch (UnknownHostException e) {
			e.printStackTrace();
		} catch (IOException e) {
			e.printStackTrace();
		}
	}
}
