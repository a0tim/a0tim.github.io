class Question{
	int a,b,c,d,e;
	public Question(){
		do{
			a = (int)(Math.random()*10);
			b = (int)(Math.random()*10);
			c = (int)(Math.random()*10);
			d = (int)(Math.random()*10);
		}while(!check(a,b,c,d));
		e = a*1000 + b*100 + c*10 + d;
		System.out.printf("解答是：%04d\n",this.e);
	}
	private boolean check(int a,int b,int c,int d){
		if (a!=b && a!=c && a!=d && b!=c && b!=d && c!=d){
			return true;
		}else{
			return false;
		}
	}
} 

class Answer{
	static void inputAns() throws IOException{
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		Question Q = new Question();
		int countA = 1;
		String Answer = "-1";
		do{
			do{
				System.out.print("請輸入第"+countA+"次的答案：");
				Answer = br.readLine();
			}while(Integer.parseInt(Answer)<0123 || Integer.parseInt(Answer)>9876);
			countA++;
		}while(!AnsCheck(Answer,Q.a,Q.b,Q.c,Q.d));
	}
	static boolean AnsCheck(String Answer,int q1,int q2,int q3,int q4){
		int Que[] = {q1,q2,q3,q4};
		int A=0,B=0;
		int Ans[] = {(int)Answer.charAt(0)-48,(int)Answer.charAt(1)-48,(int)Answer.charAt(2)-48,(int)Answer.charAt(3)-48};
		for(int i=0;i<4;i++){
			for(int j=0;j<4;j++){
				if(Ans[i] == Que[j]){
					if(i == j){
						A++;
					}else{
						B++;
					}
				}
			}
		}
		System.out.println(A+"A "+B+"B");
		if(A==4){
			return true;
		}else{
			return false;
		}
	}
}
