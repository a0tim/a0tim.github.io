import java.io.*;
import java.util.*;
public class Text1203 {
	public static void main(String[] args)throws IOException{
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		System.out.println("請輸入一行字，將以空白做為切割點切割。");
		String STR = br.readLine();
		Cut c = new Cut();
		c.Scut(STR);
		c.Sprint();
	}
}

class Cut{
	ArrayList STL = new ArrayList();
	public void Scut(String s){
		StringTokenizer STN = new StringTokenizer(s,"真理");
		do{
			STL.add(STN.nextToken());
		}while(STN.hasMoreTokens());
	}
	public void Sprint(){
		for(int i=0;i<STL.size();i++){
			System.out.println(STL.get(i));
		}
	}
}
