import java.io.BufferedReader;
import java.io.DataInputStream;
import java.io.DataOutputStream;
import java.io.IOException;
import java.io.InputStreamReader;
import java.net.ServerSocket;
import java.net.Socket;

public class Text0110_DStream_S {
	public static void main(String[] args) {
		try {
			System.out.println("Waiting for connection......");
			ServerSocket SSC = new ServerSocket(12345);
			Socket SC = SSC.accept();
			System.out.println(SC.getInetAddress()+"connect!!!");
			DataInputStream DI = new DataInputStream(SC.getInputStream());
			DataOutputStream DO = new DataOutputStream(SC.getOutputStream());
			BufferedReader BR = new BufferedReader(new InputStreamReader(System.in));
			DO.writeUTF("Welcome to server!!!");
			DO.writeUTF("Please insert your name:");
			String ClientName = DI.readUTF();
			System.out.println("Client's Name is: "+ClientName);
			DO.writeUTF("Start conversation!");
			System.out.println("Start conversation!");
			while(true){
				System.out.print("Server:");
				DO.writeUTF(BR.readLine());
				System.out.println(ClientName+":"+DI.readUTF());
			}
		} catch (IOException e) {
			e.printStackTrace();
		}
	}
}
