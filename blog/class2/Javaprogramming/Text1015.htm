import java.io.*;		//匯入IO函式庫

public class Text1015{
	public static void main(String args[]){		
		Console CS = System.console();		//定義CS為Console
		char[] b1;		//宣告字元陣列b1用來儲存驗證密碼
		Boolean end = false;	//宣告布林變數end當作迴圈中止條件
		
		System.out.println("\t======註冊======");
		System.out.print("\t-請設定帳號：");
		String a = CS.readLine();
		char[] b = CS.readPassword("\t-請設定密碼：");	//宣告字元陣列b並用readPassword的方式儲存密碼
		
		do{
			System.out.println("\n\t======驗證======");
			System.out.print("\t-請輸入帳號：");
			String a1 = CS.readLine();
			b1 = CS.readPassword("\t-請輸入密碼：");
			if(a.equals(a1)){		//檢查帳號
				if(new String(b).equals(new String(b1))){		//檢查密碼
					end = true;		//設定終止變數end為true
					System.out.println("\t！！登入成功！！");
				}else{
					System.out.println("\t！！帳號正確，密碼錯誤，登入失敗！！");
				}
			}else{
				System.out.println("\t！！帳號錯誤，登入失敗！！");
			}
		}while(!end);		//終止條件
	}
}
