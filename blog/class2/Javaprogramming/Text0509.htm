public class Text0509 {
	public static void main(String[] args) {
		Boolean Man[] = {false,false,false,false,false,false};
		Boolean hasAns = false;
		String S;
		for(int i=0;i<Math.pow(2,6);i++){
			S = Integer.toBinaryString(i);
			if(S.length()<6){\
				int X = S.length();
				for(int k=0;k<6-X;k++){
					S = "0"+S;
				}
			}
			for(int j=0;j<6;j++){
				if(S.charAt(j) == '1'){
					Man[j] = true;
				}else{
					Man[j] = false;
				}
			}
			if(Man[0]||Man[1]){
				if(Man[0]&&Man[4] || Man[0]&&Man[5] || Man[4]&&Man[5]){
					if(!(Man[0]&&Man[3])){
						if(Man[1]&&Man[2] || (!Man[1]&&!Man[2])){
							if(!Man[2]&&Man[3] || Man[2]&&!Man[3]){
								if(Man[3] || !Man[4]){
									hasAns = true;
									break;
								}
							}
						}
					}
				}
			}
		}
		if(hasAns){
			System.out.println("A:"+(Man[0]?"犯人":"不是犯人")+"\nB:"+(Man[1]?"犯人":"不是犯人")+"\nC:"+(Man[2]?"犯人":"不是犯人")+"\nD:"+(Man[3]?"犯人":"不是犯人")+"\nE:"+(Man[4]?"犯人":"不是犯人")+"\nF:"+(Man[5]?"犯人":"不是犯人"));
		}else{
			System.out.println("Have No Answers!!");
		}
	}
}
