import java.util.Scanner;	//匯入Scanner方法

public class Text1001{
        public static void main(String args[]){
		Scanner SC = new Scanner(System.in);	//宣告新Scanner叫SC

                System.out.print("請輸入第一個整數:");
		int a = SC.nextInt();		//宣告第一個數a並讓使用者輸入其值
		System.out.println("輸入:"+a);		//顯示輸入的數a
                System.out.print("請輸入第二個整數:");
		int b = SC.nextInt();		//宣告第二個數b並讓使用者輸入其值
		System.out.println("輸入:"+b);		//顯示輸入的數b
                System.out.print("請輸入第三個整數:");
		int c = SC.nextInt();		//宣告第三個數c並讓使用者輸入其值
		System.out.println("輸入:"+c);		//顯示輸入的數c
		System.out.println("三個數的加總是:"+(a+b+c));
        }
}
