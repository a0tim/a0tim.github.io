import java.io.*;

public class Text0312 {
	public static void main(String args[]){
		BufferedReader SBR = new BufferedReader(new InputStreamReader(System.in));
		File F = new File("C:\\Documents and Settings\\Administrator\\桌面\\000.txt");
		String BUFFER = "",PRINT = "";
		int Choose = 0;
		Boolean exit = false;
		
		while(true){
			System.out.println("\n請輸入要執行的內容：0.離開\n1.Buffer讀檔\t2.Buffer寫檔\n3.Stream讀檔\t4.Stream寫檔");
			try {
				Choose = Integer.parseInt(SBR.readLine());
				switch(Choose){
					case 1:
						FileReader FR = new FileReader(F);
						BufferedReader BR = new BufferedReader(FR);
						while((BUFFER = BR.readLine()) != null){
							PRINT += BUFFER;
						}
						System.out.print("使用Buffer讀檔完成!   按下Enter列印出檔案內容...");
						SBR.readLine();
						System.out.println(PRINT);
						BR.close();
						FR.close();
						PRINT = "";
						BUFFER = "";
						break;
					case 2:
						FileWriter FW = new FileWriter(F);
						BufferedWriter BW = new BufferedWriter(FW);
						System.out.println("請選擇寫檔方式：\n1.完全覆蓋\t2.寫入檔案後方");
						Choose = Integer.parseInt(SBR.readLine());
						switch(Choose){
							case 1:
								System.out.print("請輸入要寫入的字串：");
								BUFFER = SBR.readLine();
								BW.write(BUFFER);
								break;
							case 2:
								System.out.print("請輸入要寫入的字串：");
								BUFFER = SBR.readLine();
								BW.write(getfile(F) + BUFFER);
								break;
						}
						BW.close();
						FW.close();
						PRINT = "";
						BUFFER = "";
						System.out.println("檔案寫入完畢!");
						break;
					case 3:
						FileInputStream FIS = new FileInputStream(F);
						BufferedReader BR1 = new BufferedReader(new InputStreamReader(FIS));
						while((BUFFER = BR1.readLine()) != null){
							PRINT += BUFFER;
						}
						System.out.print("使用Stream讀檔完成!   按下Enter列印出檔案內容...");
						SBR.readLine();
						System.out.println(PRINT);
						BR1.close();
						FIS.close();
						PRINT = "";
						BUFFER = "";
						break;
					case 4:
						FileOutputStream FOS = new FileOutputStream(F);
						BufferedWriter BW1 = new BufferedWriter(new OutputStreamWriter(FOS));
						System.out.println("請選擇寫檔方式：\n1.完全覆蓋\t2.寫入檔案後方");
						Choose = Integer.parseInt(SBR.readLine());
						switch(Choose){
							case 1:
								System.out.print("請輸入要寫入的字串：");
								BUFFER = SBR.readLine();
								BW1.write(BUFFER);
								break;
							case 2:
								System.out.print("請輸入要寫入的字串：");
								BUFFER = SBR.readLine();
								BW1.write(getfile(F) + BUFFER);
								break;
						}
						BW1.close();
						FOS.close();
						PRINT = "";
						BUFFER = "";
						System.out.println("檔案寫入完畢!");
						break;
					case 0:
						exit = true;
				}
			} catch (IOException e) {
				e.printStackTrace();
			}
			if(exit){
				break;
			}
		}
	}
	static String getfile(File F){
		String BUFFER = "";
		BufferedReader BR;
		try {
			BR = new BufferedReader(new FileReader(F));
			while((BUFFER += BR.readLine()) != null){
			}
		} catch (FileNotFoundException e) {
			e.printStackTrace();
		} catch (IOException e) {
			e.printStackTrace();
		}
		return BUFFER + "\n";
	}
}
