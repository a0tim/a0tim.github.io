import java.io.*;
public class Text1022_2{
	public static void main(String args[])throws IOException{
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		System.out.print("請輸入A的分數：");
		int A = Integer.parseInt(br.readLine());
		System.out.print("請輸入B的分數：");
		int B = Integer.parseInt(br.readLine());
		System.out.print("請輸入C的分數：");
		int C = Integer.parseInt(br.readLine());
		String As = (A >= 60)?"PASS":"FAIL";
		String Bs = (B >= 60)?"PASS":"FAIL";
		String Cs = (C >= 60)?"PASS":"FAIL";
		switch((A/10)){
			case 10:
				As += "\t\tS";
				break;
			case 9:
				As += "\t\tA";
				break;
			case 8:
				As += "\t\tB";
				break;
			case 7:
				As += "\t\tC";
				break;
			case 6:
				As += "\t\tD";
				break;
			default:
				As += "\t\tE";
				break;
		}
		switch((B/10)){
			case 10:
				Bs += "\t\tS";
				break;
			case 9:
				Bs += "\t\tA";
				break;
			case 8:
				Bs += "\t\tB";
				break;
			case 7:
				Bs += "\t\tC";
				break;
			case 6:
				Bs += "\t\tD";
				break;
			default:
				Bs += "\t\tE";
				break;
		}
		switch((C/10)){
			case 10:
				Cs += "\t\tS";
				break;
			case 9:
				Cs += "\t\tA";
				break;
			case 8:
				Cs += "\t\tB";
				break;
			case 7:
				Cs += "\t\tC";
				break;
			case 6:
				Cs += "\t\tD";
				break;
			default:
				Cs += "\t\tE";
				break;
		}
		System.out.println("\n\t人名\t是否及格\t分數級距");
		System.out.println("\tA"+"\t"+As);
		System.out.println("\tB"+"\t"+Bs);
		System.out.println("\tC"+"\t"+Cs);
	}
}
