import java.io.*;
public class Text1025{
	public static void main(String args[])throws IOException{
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		int a[] = new int[10];
		for(int i=0;i<10;i++){
			System.out.print("請輸入第"+(i+1)+"個數：");
			a[i] = Integer.parseInt(br.readLine());
		}
		System.out.print("未排序前的陣列內容：");
		for(int i=0;i<10;i++){
		System.out.print("--"+a[i]);
		}
		System.out.println();
		int temp = 0;
		for(int i=0;i<9;i++){
			for(int j=0;j<9-i;j++){
				if(a[j] > a[j+1]){
					temp = a[j];
					a[j] = a[j+1];
					a[j+1] = temp;
				}
			}
		}
		System.out.print("已排序後的陣列內容：");
		for(int i=0;i<10;i++){
		System.out.print("--"+a[i]);
		}
	}
}
