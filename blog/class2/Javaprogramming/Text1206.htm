import java.io.*;
import java.util.StringTokenizer;
public class Text1206 {
	public static void main(String[] args)throws IOException{
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		System.out.println("請輸入要開啟的完整檔案位置及檔名附檔名：");
		String Fname = br.readLine();
		BufferedReader Fbr = new BufferedReader(new FileReader(Fname));
        String Str = new String();
        String S = new String();
		while((Str=Fbr.readLine())!=null){
			S += Str+"\n";
		}
		Fbr.close();
        System.out.println("文件內容：\n====================================\n"
        					+S+"\n====================================");
        System.out.println("\n請輸入要尋找的關鍵字：");
		String c = br.readLine();
        int KeyCount = Search.Run(S,c);
        int Slength = S.length()/2;
		System.out.println("總共出現："+KeyCount+"次");
		System.out.println("總字數為："+Slength+"個");
		System.out.println("重要性為："+KeyCount+"/"+Slength+" = "+(float)KeyCount/Slength*100+"%");
		System.out.println("分別出現在：\n====================================\n"
        					+Search.show()+"\n====================================");
		System.out.println("\n\nThe End Of Program！");
		br.close();
	}
}

class Search{
	private static int count = 0;
	private static String S_c = new String();
	private static String C_l = new String();
	public static int Run(String S,String c){
		StringTokenizer STN = new StringTokenizer(S,c);
		Cstar(c);
		while(STN.hasMoreTokens()){
			if(count==0){
				S_c += STN.nextToken();
				count ++;
			}else{
				S_c += C_l+STN.nextToken();
				count ++;
			}
		}
		return --count;
	}
	private static String Cstar(String c){
		for(int i=0;i<c.length();i++){
			C_l += "●";
		}
		return C_l;
	}
	public static String show(){
		return S_c;
	}
}
